-- Strip rich text tags, normalize, trim
local function stripTags(s) return type(s)=="string" and s:gsub("<.->","") or "" end
local function normalize(s) return type(s)=="string" and s:lower():gsub("%s+",""):gsub("[^%a%d]","") or "" end

local function isInBadService(obj)
    local cur = obj.Parent
    for _ = 1, 16 do
        if not cur or cur == game then return false end
        local n = cur.Name
        if n=="ReplicatedStorage" or n=="ServerStorage" or n=="PlayerGui"
        or n=="StarterGui" or n=="Lighting" or n=="CoreGui"
        or n=="StarterPack" or n=="StarterPlayer" then return true end
        cur = cur.Parent
    end
    return false
end

local function isInAnyChar(obj)
    for _, p in ipairs(Players:GetPlayers()) do
        if p.Character and obj:IsDescendantOf(p.Character) then return true end
    end
    return false
end

local function getPhysPart(obj)
    if obj:IsA("BasePart") then return obj end
    if obj:IsA("Model") then
        if obj.PrimaryPart then return obj.PrimaryPart end
        for _, c in ipairs(obj:GetChildren()) do if c:IsA("BasePart") then return c end end
        for _, c in ipairs(obj:GetDescendants()) do if c:IsA("BasePart") then return c end end
    end
    return nil
end

local BALL_CACHE = {}; local BALL_SET = {}; local SCAN_DONE = false; local selectedBallIdx = nil
local function hasball(n) return n:lower():find("ball",1,true) ~= nil end
local function isClientBall(n) return n:upper():sub(1,12) == "CLIENT_BALL_" end

local function tryAdd(obj)
    if BALL_SET[obj] then return end
    if not hasball(obj.Name) then return end
    if BOLT_OWN[obj.Name] then return end
    if isInBadService(obj) then return end
    if isInAnyChar(obj) then return end
    local part = getPhysPart(obj); if not part then return end
    local e = {part=part, name=obj.Name, obj=obj}
    BALL_SET[obj] = e; table.insert(BALL_CACHE, e)
end

local function tryRemove(obj)
    if not BALL_SET[obj] then return end
    BALL_SET[obj] = nil
    local new = {}
    for _, e in ipairs(BALL_CACHE) do if e.obj ~= obj then table.insert(new, e) end end
    BALL_CACHE = new
    if selectedBallIdx and selectedBallIdx > #BALL_CACHE then
        selectedBallIdx = #BALL_CACHE > 0 and #BALL_CACHE or nil
    end
end

local pTick = 0
local function pruneCache()
    pTick = pTick + 1; if pTick % 15 ~= 0 then return end
    local new = {}; local ns = {}
    for _, e in ipairs(BALL_CACHE) do
        if e.part and e.part.Parent and not isInAnyChar(e.part) then
            table.insert(new, e); ns[e.obj] = e
        else BALL_SET[e.obj] = nil end
    end
    BALL_CACHE = new; BALL_SET = ns
    if selectedBallIdx and selectedBallIdx > #BALL_CACHE then
        selectedBallIdx = #BALL_CACHE > 0 and #BALL_CACHE or nil
    end
end

local function getBV(entry)
    local best = Vector3.new(0,0,0)
    pcall(function() best = entry.part.AssemblyLinearVelocity end)
    return best
end

local function ballScore(e)
    if not (e.part and e.part.Parent) then return -1 end
    local sc = 0; local vel = getBV(e)
    sc = sc + math.min(vel.Magnitude*5, 150)
    if not e.part.Anchored then sc = sc + 80 end
    if isClientBall(e.name) then sc = sc + 1000 end
    pcall(function() if e.part:IsA("Part") and e.part.Shape == Enum.PartType.Ball then sc = sc + 40 end end)
    local mx = math.max(e.part.Size.X, e.part.Size.Y, e.part.Size.Z)
    if mx <= 4 then sc = sc + 30 end; if mx <= 2 then sc = sc + 20 end
    return sc
end

local function getSelectedBall()
    pruneCache(); if #BALL_CACHE == 0 then return nil end
    if selectedBallIdx ~= nil then
        return BALL_CACHE[math.clamp(selectedBallIdx, 1, #BALL_CACHE)]
    end
    local best, bestSc = BALL_CACHE[1], -math.huge
    for _, e in ipairs(BALL_CACHE) do
        local sc = ballScore(e); if sc > bestSc then bestSc = sc; best = e end
    end
    return best
end

local function cycleBall()
    pruneCache(); if #BALL_CACHE == 0 then selectedBallIdx = nil; return nil end
    selectedBallIdx = ((selectedBallIdx or 0) % #BALL_CACHE) + 1
    return BALL_CACHE[selectedBallIdx]
end

local GARBAGE_PATS = {"^%d+$","^%d+%%$","^rbxassetid://%d+$","^lvl$","^level$","^lv$","^rank$","^tier$","^equip$","^select$","^unlock$","^lock$","^u$","^x$","^v$","^ok$","^go$","^on$","^off$","^yes$","^no$","^up$","^dn$","^new$","^old$","^atk$","^def$","^spd$","^hp$","^mp$","^xp$","^cost$","^price$","^buy$","^sell$","^equipped$","^unequipped$","^locked$","^unlocked$"}
local function isGarbage(text)
    if not text or type(text) ~= "string" then return true end
    local t = trim(text):lower(); if #t < 3 or #t > 30 then return true end
    if t:match("^rbxassetid://") then return true end
    for _, p in ipairs(GARBAGE_PATS) do if t:match(p) then return true end end
    return #(t:gsub("%d",""):gsub("%%","")) < 2
end
local function isVSN(text)
    if not text or type(text) ~= "string" then return false end
    local t = trim(text); if #t < 3 or #t > 30 or isGarbage(t) then return false end
    return #(t:gsub("%d",""):gsub("%s",""):gsub("[^%a]","")) >= 2
end
local function isVAN(text)
    if not text or type(text) ~= "string" then return false end
    local t = trim(text); if #t < 3 or #t > 40 then return false end
    if t:match("^%d+$") or t:match("^rbxassetid://") then return false end
    local bad = {"equipped","unequipped","locked","unlocked","unlock","equip","select","common","rare","epic","legendary","slot"}
    local tl = t:lower(); for _, w in ipairs(bad) do if tl == w then return false end end
    return #(t:gsub("[^%a%s]","")) >= 2
end

local function getCleanText(inst)
    local ok1, ct = pcall(function() return inst.ContentText end)
    if ok1 and ct and #ct > 0 then local cl = trim(ct); if isVSN(cl) then return cl end end
    local ok2, t = pcall(function() return inst.Text end)
    if ok2 and t and #t > 0 then local cl = trim(stripTags(t)); if isVSN(cl) then return cl end end
    return nil
end

local function extractText(inst)
    pcall(function()
        if inst:IsA("TextLabel") or inst:IsA("TextButton") or inst:IsA("TextBox") then
            local tx = getCleanText(inst); if tx then return tx end
        end
    end)
    return nil
end

local function getStyleFromChar()
    local styles, si = {}, {}
    pcall(function()
        local char = LocalPlayer.Character; if not char then return end
        local jb = char:FindFirstChild("JerseyBack")
        if jb then
            for _, c in pairs(jb:GetDescendants()) do
                local tx = extractText(c)
                if tx then table.insert(styles,{text=tx,normalized=normalize(tx),equipped=true}); table.insert(si,"JB:"..tx) end
            end
        end
        local nt = char:FindFirstChild("Nametag")
        if nt then
            local cont = nt:FindFirstChild("Container")
            if cont then
                for _, c in pairs(cont:GetChildren()) do
                    local nl = c.Name:lower()
                    if nl:match("stylename") or nl:match("name") then
                        local tx = extractText(c)
                        if tx then table.insert(styles,{text=tx,normalized=normalize(tx),equipped=true}); table.insert(si,"NT:"..tx) end
                    end
                end
            end
        end
    end)
    return styles, si
end

local function getCurrentStyle()
    local styles, si = getStyleFromChar()
    pcall(function()
        for _, g in pairs(PlayerGui:GetChildren()) do
            if g.Name:lower():match("style") or g.Name:lower():match("inventory") then
                for _, c in pairs(g:GetDescendants()) do
                    local tx = extractText(c)
                    if tx then table.insert(styles,{text=tx,normalized=normalize(tx),equipped=false}); table.insert(si,"GUI:"..tx) end
                end
            end
        end
    end)
    if #styles > 0 then
        for _, s in ipairs(styles) do if s.equipped then return s.text, s.normalized, styles end end
        return styles[1].text, styles[1].normalized, styles
    end
    return "Unknown", "", {}
end

local function checkHasStyle(n, ss)
    local s = normalize(n); if s == "" or not ss or #ss == 0 then return false end
    for _, item in ipairs(ss) do if item.normalized == s then return true end end
    return false
end

end

local KeySystem = {}

function KeySystem:Save(key)
    pcall(function()
        if writefile then writefile(KEY_FILE, HttpService:JSONEncode({key=key,ts=os.time()})) end
    end)
end

function KeySystem:Load()
    local ok, data = pcall(function()
        if isfile and isfile(KEY_FILE) then
            return HttpService:JSONDecode(readfile(KEY_FILE))
        end
        return nil
    end)
    if ok and data and data.key then return data.key end
    return nil
end

function KeySystem:Validate(key, callback)
    task.spawn(function()
        local ok, result = pcall(function() return game:HttpGet(KEY_API) end)
        if not ok then callback(false, "Connection failed"); return end
        local validKeys = {}
        local jok, decoded = pcall(function() return HttpService:JSONDecode(result) end)
        if jok and type(decoded) == "table" then
            if decoded.keys and type(decoded.keys) == "table" then
                validKeys = decoded.keys
            else
                for _, v in pairs(decoded) do
                    if type(v) == "string" then table.insert(validKeys, v)
                    elseif type(v) == "table" and v.key then table.insert(validKeys, tostring(v.key)) end
                end
            end
        else
            for line in result:gmatch("[^\r\n]+") do
                line = line:match("^%s*(.-)%s*$")
                if #line > 0 then table.insert(validKeys, line) end
            end
        end
        local clean = key:match("^%s*(.-)%s*$")
        local valid = false
        for _, vk in ipairs(validKeys) do
            if tostring(vk):lower() == clean:lower() then valid = true; break end
        end
        if valid then self:Save(clean); callback(true, "Key validated") else callback(false, "Invalid key") end
    end)
end

    local function doStyleSpin()
        if styleSpinning then return end
        styleSpinning = true; local rolls = 0
        autoStyleBtn.Text = " STOP SPINNING"
        tw(autoStyleBtn,{BackgroundColor3=T.Danger},0.2)

        while styleSpinning do
            rolls = rolls + 1
            styleCountLbl.Text = "Rolls: " .. rolls .. "  |  Last: —"
            styleLbl.Text = "Rolling... (#" .. rolls .. ")"
            styleLbl.TextColor3 = T.Gold; tw(styleDot,{BackgroundColor3=T.Gold},0.1)

            local targets = parseTargets(styleInput.Text)
            local _, ss = updateStyleDisplay()
            local pre = nil
            for _, t in ipairs(targets) do if checkHasStyle(t, ss) then pre = t; break end end
            if pre then
                styleSpinning = false
                styleLbl.Text = " Already have " .. pre .. "!"
                styleLbl.TextColor3 = T.Success; tw(styleDot,{BackgroundColor3=T.Success},0.2)
                notify("Already have " .. pre .. "!", T.Success); break
            end

            local ok = pcall(invokeStyleRoll)
            local found = nil
            for _ = 1, 15 do
                task.wait(0.04)
                local _, ssa = updateStyleDisplay()
                for _, t in ipairs(targets) do if checkHasStyle(t, ssa) then found = t; break end end
                if found then break end
            end
            if not found then
                task.wait(0.12)
                local _, ssa = updateStyleDisplay()
                for _, t in ipairs(targets) do if checkHasStyle(t, ssa) then found = t; break end end
            end

            if ok then
                if found then
                    styleSpinning = false
                    styleLbl.Text = " " .. found .. " OBTAINED!"
                    styleLbl.TextColor3 = T.Success; tw(styleDot,{BackgroundColor3=T.Success},0.2)
                    notify(" " .. found .. " OBTAINED!", T.Success); break
                else
                    local st = getCurrentStyle()
                    styleCountLbl.Text = "Rolls: " .. rolls .. "  |  Last: " .. st:sub(1,18)
                    styleLbl.Text = "Got: " .. st:sub(1,22)
                    styleLbl.TextColor3 = T.Gold; tw(styleDot,{BackgroundColor3=T.Gold},0.1)
                end
            else
                styleLbl.Text = "Roll failed"; styleLbl.TextColor3 = T.Error
            end
            task.wait(0.05)
        end

        autoStyleBtn.Text = " START STYLE SPIN"
        tw(autoStyleBtn,{BackgroundColor3=T.Accent},0.2)
        if not styleSpinning then
            styleLbl.TextColor3 = T.SubText; tw(styleDot,{BackgroundColor3=T.SubText},0.2)
        end
        styleSpinning = false
    end
